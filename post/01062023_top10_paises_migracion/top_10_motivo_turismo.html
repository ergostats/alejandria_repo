ggplot_df <- function(tabla){
  
  niveles <- tabla %>% 
    pull(homologada) %>% 
    unique()

  tabla <- tabla %>% 
    mutate(transp = 1)
  
  size <- length(niveles)
  
  if(size != 10){
    
    nuevos <- str_c("Ficticio",1:(10-size))
    
    tabla <- tibble(homologada = nuevos,
                    salidas = 0.5,
                    transp = 0) %>% 
      bind_rows(tabla)
    
    
    colores_text <- c(rep("white",10 - size),rep("darkgrey",size))
    
  } else{
    colores_text <- rep("darkgrey",10)
  }
  
  # browser()
  
  tabla <- tabla %>% 
    mutate(
      homologada = factor(homologada),
      homologada = fct_reorder(homologada,rank),
      homologada = fct_rev(homologada),
      colores = colores_text
    ) 

  
  return(tabla)
  
  
}

dates_treat <- function(fecha_centro,
                        date_df){

  # browser()
  
  date_df <- date_df %>% 
    arrange(dias)
  
 objetivos <- date_df %>% 
   ungroup() %>% 
   mutate(izquierda = fecha_centro - months(7),
          derecha = fecha_centro + months(7),
          interval = lubridate::interval(izquierda,
                              derecha)) %>% 
   filter(dias %within% interval)
 
 remanente <- date_df %>% 
   anti_join(objetivos) %>% 
   mutate(altura = 0.05) 
 
 # browser()
 
 secuencia <- tibble(
   x = seq(-2.1,2.1,by =  2.1*2/14),
   dias = seq(from = (fecha_centro - months(7)),
                to = (fecha_centro + months(7)),
                by = "month"),
   altura = dnorm(x)
 )
   
 
 
 
 
 objetivos %>% 
   left_join(secuencia) %>% 
   bind_rows(remanente) %>%
   mutate(centro = fecha_centro) %>% 
   arrange(dias) 
 
  
}
df_gif <- date_table %>%
  pull(dias) %>% 
  map(~dates_treat(fecha_centro = .x,
                   date_df = date_table)) %>%

plots <- df_gif  %>%
